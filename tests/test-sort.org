from vasp import Vasp
from ase.lattice.surface import fcc111
import logging
log = logging.getLogger('Vasp')
#log.setLevel(logging.DEBUG)

# Create surface image
atoms = fcc111('Cu', size=(3, 3, 5), vacuum=6.0, a=3.628)
atoms[40].symbol = 'Pd'

calc = Vasp('impurity-calc',
	    xc='PBE',
	    kpts=[2, 2, 1],
	    encut=250,
	    atoms=atoms)
calc.write_input()

print(atoms[40])
